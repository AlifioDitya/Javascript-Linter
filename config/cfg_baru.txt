S -> ASSIGNMENT | IF_STATEMENT | COMMENT | epsilon
IF_STATEMENT -> if ( CONDITIONAL ) { IF_BLOCK } | if ( CONDITIONAL ) { IF_BLOCK } ELSE_STATEMENT | IF_STATEMENT S
IF_BLOCK -> S | S IF_BLOCK
ELSE_STATEMENT -> else IF_STATEMENT | else { IF_BLOCK }
ASSIGNMENT -> IDENTIFIER ASSIGN_OP DATA | ASSIGNMENT COMMENT | ASSIGNMENT S
ASSIGN_OP -> = | += | -= | *= | /= | %= | **= | LOGICAL_OP = | BITWISE_OP =
COMPARISON_OP -> == | === | != | !== | > | < | >= | <=
LOGICAL_OP -> && | ||
BITWISE_OP -> & | | | ~ | ^ | << | >> | >>>
COMMENT -> SINGLE_COMMENT | MULTI_COMMENT
SINGLE_COMMENT -> // SINGLE_COMMENT_CONTENT
SINGLE_COMMENT_CONTENT -> ANY | ANY SINGLE_COMMENT_CONTENT
MULTI_COMMENT -> /* MULTI_COMMENT_CONTENT */
MULTI_COMMENT_CONTENT -> ANY | ANY MULTI_COMMENT_CONTENT | 
CONDITIONAL -> DATA | DATA LOGICAL_OP DATA | DATA COMPARISON_OP DATA | ! CONDITIONAL | CONDITIONAL LOGICAL_OP CONDITIONAL | ( CONDITIONAL )
IDENTIFIER -> @identifier | @identifier . @identifier | LIST_OP
DATA -> TERNARY_DATA | NONTERNARY_DATA
TERNARY_DATA -> CONDITIONAL ? DATA : DATA
NONTERNARY_DATA -> STRING | @integer | FLOAT | LIST | TUPLE | SET | BOOLEAN | IDENTIFIER | CONDITIONAL | MATH_OPERATION | FUNCTION_CALL | None | LIST_OPER | LIST_SLICE
STRING -> " ANY " | ' ANY ' | " " | ' '
FLOAT -> @integer . | @integer . @integer
LIST -> [ ] | [ LIST_DATA ]
LIST_DATA -> DATA | DATA , LIST_DATA
ANY -> @any | @any ANY
