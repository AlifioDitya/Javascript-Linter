S -> IF ENDLINE | RETURN ENDLINE | epsilon
IF -> if ( EXPRESSION ) { S } ENDLINE | IF ( EXPRESSION ) { S } ELSE
ELSE -> else { S } ENDLINE S | else IF
EXPRESSION -> BOOLEAN | ( BOOLEAN ) | BRACKET
BOOLEAN -> true ENDLINE | false ENDLINE | @varval @operator @varval
BRACKET -> ( @varval ) ENDLINE | ( ) ENDLINE
ENDLINE -> ; ENDLINE | epsilon
RETURN -> return @varval ENDLINE | return ENDLINE
